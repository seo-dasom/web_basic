선택박스로 현재년 부터 ~ 1970년 까지의 옵션 생성하기.

<select id="exec1" onfocus="exercise1(this);"> <option value="year">년도</option> </select>

function exercise1(node) { let date = new Date(); if (node.children.length <= 1) { for(let year = date.getFullYear(); year >= 1970; year--) { let option = document.createElement("option"); option.setAttribute("value", year); option.innerText = year; node.appendChild(option); } } }

function exercise1(node) { let date = new Date(); if (node.children.length <= 1) { for(let year = date.getFullYear(); year >= 1970; year--) { let option = $("<option></option>") option.attr("value", year); option.text(year); $("#"+node.id).append(option); } } }
